 11    adata_CG,
 12    adata_CP,
 13    model_dir,
 14    *,
 15    snps=None,
 16    regions=None,
 17    window_size=500_000,
 18    celltype_specific=False,
 19    skip_uncertain=True,
 20    use_distance_weight=False,
 21):
 22    """
 23    Unsupervised SIMBA+ peak–gene link prediction.
 24
 25    Performs the full SIMBA+ inference pipeline:
 26      1. Compute overlaps between SNPs/regions and ATAC peaks.
 27      2. Form candidate peak–gene pairs within a cis window.
 28      3. Compute SIMBA+ path scores for each pair.
 29      4. Attach SNP/region annotations (if provided).
 30
 31    Parameters
 32    ----------
 33    adata_CG : anndata.AnnData
 34        Gene-level SIMBA+ AnnData containing gene embeddings and metadata.
 35
 36    adata_CP : anndata.AnnData
 37        Peak-level SIMBA+ AnnData containing peak embeddings and metadata.
 38
 39    model_dir : str
 40        Directory containing trained SIMBA+ outputs:
 41            - adata_C.h5ad
 42            - adata_G.h5ad
 43            - adata_P.h5ad
 44
 45    snps : pandas.DataFrame, optional
 46        DataFrame with SNP locations.
 47        Required columns: ["snpid", "chr", "pos"].
 48        If None, SNP scoring is skipped.
 49
 50    regions : pandas.DataFrame, optional
 51        DataFrame of genomic regions.
 52        Required columns: ["regionid", "chr", "start", "end"].
 53
 54    window_size : int, default 500000
 55        Genomic distance (bp) for forming candidate peak–gene pairs.
 56
 57    celltype_specific : bool, default False
 58        If True, compute separate SIMBA+ scores per cell type.
 59
 60    skip_uncertain : bool, default True
 61        Exclude cells with uncertain/unknown annotations.
 62
 63    use_distance_weight : bool, default False
 64        Multiply SIMBA+ scores by (1 / distance_to_TSS).
 65
 66    Returns
 67    -------
 68    pandas.DataFrame
 69        Peak–gene link table containing:
 70        - Peak
 71        - Gene_name
 72        - SIMBA+ features
 73        - optional SNP/region metadata
 74        - Distance_to_TSS
 75        - additional covariates produced by add_simba_plus_features()
 76
 77    Notes
 78    -----
 79    This function **does not** require a supervised model and performs purely
 80    unsupervised SIMBA+ scoring.
 81
 82    """
 83
 84    # 1. Overlaps
 85    region_hits, region_meta = find_region_peak_overlaps(regions, adata_CP)
 86    snp_hits, snp_meta = find_snp_peak_overlaps(snps, adata_CP)
 87
 88    peak_like = build_candidate_peaks(adata_CP, snp_hits, region_hits)
 89
 90    # 2. Candidate peak–gene links
 91    links = pgl.get_peak_gene_links(
 92        peaks=peak_like,
 93        genes=list(adata_CG.var_names),
 94        cis_window=window_size,
 95    )
 96
 97    if links.empty:
 98        print("No candidate peak–gene links found.")
 99        return links
100
101    # 3. Add SIMBA+ path scores
102    links = score.add_simba_plus_features(
103        eval_df=links,
104        adata_C_path=os.path.join(model_dir, "adata_C.h5ad"),
105        adata_G_path=os.path.join(model_dir, "adata_G.h5ad"),
106        adata_P_path=os.path.join(model_dir, "adata_P.h5ad"),
107        gene_col="Gene_name",
108        peak_col="Peak",
109        celltype_specific=celltype_specific,
110        skip_uncertain=skip_uncertain,
111        use_distance_weight=use_distance_weight,
112    )
113
114    # 4. Attach SNP/region annotations
115    if snp_meta is not None:
116        links = snp_meta.merge(links, on="Peak", how="left")
117    if region_meta is not None:
118        links = region_meta.merge(links, on="Peak", how="left")
119
120    return links

123def load_crispr_eval(

163    crispr_input.to_csv(output_path, sep="\t", index=False)

304    # Load AnnData

324
